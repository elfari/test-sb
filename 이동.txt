    <div class="container">
        <textarea id="textArea" placeholder="여기에 텍스트를 입력하세요..."></textarea>
        
        <div class="controls">
            <select id="userSelect">
                <option value="A">User A</option>
                <option value="B">User B</option>
            </select>

            <button id="btnLoad" onclick="loadText()">불러오기</button>

            <button id="btnSend" onclick="sendText()">전송</button>
        </div>
    </div>

    <script>
        const textArea = document.getElementById('textArea');
        const userSelect = document.getElementById('userSelect');
        let lastLoadedText = "";

        // [기능 4 & 필수요건] 텍스트 전송 (AJAX) - (이전과 동일)
        function sendText() {
            const text = textArea.value;
            const user = userSelect.value;

            fetch('clipboard.jsp?action=save', {
                method: 'POST',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
                body: 'user=' + encodeURIComponent(user) + '&text=' + encodeURIComponent(text)
            })
            .then(() => {
                console.log("전송 완료");
                lastLoadedText = text;
            });
        }

        // [기능 3] 텍스트 불러오기 (AJAX) - (이전과 동일)
        function loadText() {
            const user = userSelect.value;

            fetch('clipboard.jsp?action=load&user=' + encodeURIComponent(user))
            .then(response => response.text())
            .then(text => {
                if (text !== lastLoadedText) {
                    textArea.value = text.trim();
                    lastLoadedText = text;
                }
            });
        }

        // ------------------------------------------------------------------
        // [수정된 부분] 엔터 키 입력 시 바로 전송 & 줄바꿈 방지
        // ------------------------------------------------------------------
        textArea.addEventListener('keydown', function(e) {
            // Enter 키만 감지
            if (e.key === 'Enter') {
                e.preventDefault(); // Enter 키의 기본 동작(줄바꿈) 막기
                sendText();         // 즉시 전송 함수 호출
            }
        });

        // [필수요건 대응] 자동 동기화 (1초마다 서버 상태 확인) - (이전과 동일)
        setInterval(loadText, 1000);

        // 사용자 변경 시 즉시 해당 사용자의 데이터 로드 - (이전과 동일)
        userSelect.addEventListener('change', () => {
            lastLoadedText = "";
            loadText();
        });

        // 초기 로드 - (이전과 동일)
        loadText();
    </script>
