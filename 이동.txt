SELECT LISTAGG(word, ' ') WITHIN GROUP (ORDER BY lvl) AS result
FROM (
    SELECT DISTINCT 
           REGEXP_SUBSTR('갤럭시 갤럭시 S25', '[^ ]+', 1, LEVEL) AS word,
           MIN(LEVEL) AS lvl
    FROM DUAL
    CONNECT BY REGEXP_SUBSTR('갤럭시 갤럭시 S25', '[^ ]+', 1, LEVEL) IS NOT NULL
    GROUP BY REGEXP_SUBSTR('갤럭시 갤럭시 S25', '[^ ]+', 1, LEVEL)
);
