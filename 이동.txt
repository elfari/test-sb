plugins: [
    Ext.create('Ext.grid.plugin.CellEditing', {
        clicksToEdit: 1,
        listeners: {
            // 편집 모드 들어가기 직전에 실행되는 이벤트
            beforeedit: function(editor, context) {
                var record = context.record;
                
                // 조건: 자식 노드(Leaf)인 경우 편집 불가 (return false로 취소)
                // 즉, 부모 노드(폴더)만 편집 가능
                if (record.isLeaf()) {
                    return false; 
                }
                
                // (선택사항) 특정 컬럼만 제어하고 싶다면?
                // if (record.isLeaf() && context.field === 'reasonCode') { return false; }
            }
        }
    })
