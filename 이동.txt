<%
    // --- [백엔드 로직] 데이터 처리 영역 (JDK 1.7 호환) ---
    request.setCharacterEncoding("UTF-8");
    String action = request.getParameter("action");
    
    // 최대 라인 수 및 삭제 비율 상수 정의
    final int MAX_LINES = 10000;
    final double DELETE_PERCENTAGE = 0.3; // 30%

    // 데이터 저장소 (서버 메모리 공유: application scope)
    if ("save".equals(action)) {
        String user = request.getParameter("user");
        String text = request.getParameter("text");
        
        if (user != null && text != null) {
            String existingText = (String) application.getAttribute("clipboard_" + user);
            
            // 1. 기존 텍스트에 새 텍스트를 추가 (줄바꿈 추가)
            String newText;
            if (existingText == null || existingText.isEmpty()) {
                newText = text;
            } else {
                // 이전 내용과 현재 전송된 내용 사이에 줄바꿈 추가 (클립보드 내용 누적)
                newText = existingText + "\n" + text; 
            }
            
            // 2. 라인 수 검사 및 삭제 로직
            // \r?\n: 모든 종류의 줄바꿈 문자로 분리
            String[] lines = newText.split("\\r?\\n"); 
            
            if (lines.length > MAX_LINES) {
                // 초과된 경우: 상위 30% 라인 수 계산
                int linesToDelete = (int) (lines.length * DELETE_PERCENTAGE);
                if (linesToDelete == 0) linesToDelete = 1; // 최소 1줄은 삭제 보장

                // 상위 linesToDelete 개를 제외한 나머지 라인만 저장
                // JDK 1.7 호환: Arrays.copyOfRange 사용
                String[] remainingLines = Arrays.copyOfRange(lines, linesToDelete, lines.length);
                
                // 3. JDK 1.7 호환: StringBuilder를 사용하여 배열을 줄바꿈으로 합치기
                StringBuilder sb = new StringBuilder();
                for (int i = 0; i < remainingLines.length; i++) {
                    if (i > 0) {
                        sb.append("\n");
                    }
                    sb.append(remainingLines[i]);
                }
                newText = sb.toString();
            }
            
            // 4. 서버에 최종 텍스트 저장
            application.setAttribute("clipboard_" + user, newText);
        }
        return; // HTML 렌더링 중단
    } else if ("load".equals(action)) {
        String user = request.getParameter("user");
        String text = (String) application.getAttribute("clipboard_" + user);
        out.print(text == null ? "" : text);
        return; // HTML 렌더링 중단
    }
%>

        </div>
        <textarea id="textArea" placeholder="여기에 텍스트를 입력하세요. 엔터(Enter)를 누르면 바로 전송됩니다."></textarea>
        
        <div class="controls">
            <select id="userSelect">
                <option value="A">User A</option>
                <option value="B">User B</option>
            </select>

            <button id="btnLoad" onclick="loadText()">불러오기</button>

            <button id="btnSend" onclick="sendText()">전송</button>

    <script>
        // --- [자바스크립트 로직] (이전과 동일 - 브라우저 기능이므로 JDK 버전에 영향 없음) ---
        const textArea = document.getElementById('textArea');
        const userSelect = document.getElementById('userSelect');
        let lastLoadedText = "";

        // 텍스트 전송 (AJAX)
        function sendText() {
            const text = textArea.value.trim(); 
            const user = userSelect.value;
            
            if (text === "") return; 

            textArea.value = ""; 

            fetch('clipboard.jsp?action=save', {
                method: 'POST',
                headers: { 'Content-Type': 'application/x-www-form-urlencoded' },
                body: 'user=' + encodeURIComponent(user) + '&text=' + encodeURIComponent(text)
            })
            .then(() => {
                console.log("전송 완료");
                loadText(true); 
            });
        }

        // 텍스트 불러오기 (AJAX)
        function loadText(forceUpdate = false) {
            const user = userSelect.value;

            fetch('clipboard.jsp?action=load&user=' + encodeURIComponent(user))
            .then(response => response.text())
            .then(text => {
                const trimmedText = text.trim();

                if (trimmedText !== lastLoadedText || forceUpdate) {
                    textArea.value = trimmedText;
                    lastLoadedText = trimmedText;
                }
            });
        }

        // 엔터 키 입력 시 바로 전송 & 줄바꿈 방지
        textArea.addEventListener('keydown', function(e) {
            if (e.key === 'Enter') {
                e.preventDefault();
                sendText();
            }
        });

        // 자동 동기화 (1초마다 서버 상태 확인)
        setInterval(loadText, 1000);

        // 사용자 변경 시 즉시 해당 사용자의 데이터 로드
        userSelect.addEventListener('change', () => {
            lastLoadedText = "";
            loadText(true);
        });

        // 초기 로드
        loadText(true);
    </script>
