-- 특정 문자열('CUSTOMER_ID')이 포함된
-- 프로시저 / 함수 / 패키지 / 트리거 / 뷰 소스 검색 (USER 스키마 기준)
SELECT 
    s.type        AS object_type,
    s.name        AS object_name,
    s.line        AS line_no,
    s.text        AS source_line
FROM user_source s
WHERE REGEXP_LIKE(s.text, 'CUSTOMER_ID', 'i')

UNION ALL

SELECT
    'VIEW'        AS object_type,
    v.view_name   AS object_name,
    NULL          AS line_no,
    DBMS_METADATA.GET_DDL('VIEW', v.view_name) AS source_line
FROM user_views v
WHERE REGEXP_LIKE(DBMS_METADATA.GET_DDL('VIEW', v.view_name), 'CUSTOMER_ID', 'i')

ORDER BY object_type, object_name, line_no;
