#Requires AutoHotkey v2.0

interval := 9 * 60 * 1000 ; 9분(밀리초)

DoAction(*) {
    ; 마우스 1px 우->좌
    MouseMove(1, 0, 0, true)
    Sleep(30)
    MouseMove(-1, 0, 0, true)
    Sleep(30)

    ; Shift 키 누름/뗌
    Send("{Shift down}")
    Sleep(50)
    Send("{Shift up}")
}

; 9분마다 실행
SetTimer(DoAction, interval)

; 시작하자마자 한 번 실행 (원치 않으면 지우세요)
DoAction()

; 종료 단축키: Ctrl+Alt+X
^!x::ExitApp
